76  inline void Filter_::computeSynthesisBlock( const SynthesisContext_ & context ){
77 
78  TonicFloat cCutoff;
79  TonicFloat cQ;
80 
81  // get cutoff and Q inputs
82  // For now only using first frame of output. Setting coefficients each frame is very inefficient.
83  // Updating cutoff every 64-samples is typically fast enough to avoid audible artifacts when sweeping filters.
84 
85  cutoff_.tick(workspace_, context);
86  cCutoff = clamp(workspace_(0,0), 20, sampleRate()/2); // clamp to reasonable range
87 
88  Q_.tick(workspace_, context);
89  cQ = max(workspace_(0,0), 0.7071); // clamp to reasonable range
90 
91  applyFilter(cCutoff, cQ, context);
92 
93  }

107  inline void applyFilter( TonicFloat cutoff, TonicFloat Q, const SynthesisContext_ & context )
108  {
109 
110  TonicFloat *inptr = &dryFrames_[0];
111  TonicFloat *outptr = &outputFrames_[0];
112  TonicFloat coef = cutoffToOnePoleCoef(cutoff);
113  TonicFloat norm = bNormalizeGain_ ? 1.0f - coef : 1.0f;
114  unsigned int nChannels = dryFrames_.channels();
115 
116  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
117  for (unsigned int c=0; c<nChannels; c++){
118  lastOut_[c] = (norm * (*inptr++)) + (coef * lastOut_[c]);
119  *outptr++ = lastOut_[c];
120  }
121  }
122  }
123 
124  public:
125 
126  LPF6_(){
127  lastOut_[0] = 0;
128  lastOut_[1] = 0;
129  }
