90  inline void Compressor_::computeSynthesisBlock(const SynthesisContext_ &context){
91 
92  // Tick all scalar parameters
93  float attackCoef = t60ToOnePoleCoef(max(0,attackGen_.tick(context).value));
94  float releaseCoef = t60ToOnePoleCoef(max(0, releaseGen_.tick(context).value));
95  float threshold = max(0,threshGen_.tick(context).value);
96  float ratio = max(0,ratioGen_.tick(context).value);
97  float lookaheadTime = max(0,lookaheadGen_.tick(context).value);
98 
99  // Absolute value of amplitude frames in prep for amp envelope
100  TonicFloat * ampData = &ampInputFrames_[0];
101 
102  #ifdef USE_APPLE_ACCELERATE
103  vDSP_vabs(ampData, 1, ampData, 1, ampInputFrames_.size());
104  #else
105  for (unsigned int i=0; i<ampInputFrames_.size(); i++, ampData++){
106  *ampData = fabsf(*ampData);
107  }
108  #endif
109 
110  // Iterate through samples
111  unsigned int nChannels = outputFrames_.channels();
112  TonicFloat ampInputValue, gainValue, gainTarget;
113  TonicFloat * outptr = &outputFrames_[0];
114  TonicFloat * dryptr = &dryFrames_[0];
115  ampData = &ampInputFrames_[0];
116 
117  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
118 
119  // Tick input into lookahead delay and get amplitude input value - max of left/right
120  ampInputValue = 0;
121  for (unsigned int i=0; i<nChannels; i++){
122  lookaheadDelayLine_.tickIn(*dryptr++, i);
123  ampInputValue = max(ampInputValue,*ampData++);
124  }
125 
126  // Smooth amplitude input
127  if (ampInputValue >= ampEnvValue_){
128  onePoleLPFTick(ampInputValue, ampEnvValue_, attackCoef);
129  }
130  else {
131  onePoleLPFTick(ampInputValue, ampEnvValue_, releaseCoef);
132  }
133 
134  // Calculate gain value
135 
136  if (ampEnvValue_ <= threshold){
137  gainValue = 1.0f;
138  }
139  else{
140  // compensate for ratio
141  gainTarget = threshold + ((ampEnvValue_ - threshold)/ratio);
142  gainValue = gainTarget/ampEnvValue_;
143  }
144 
145  // Smooth gain value
146  if (gainValue <= gainEnvValue_){
147  onePoleLPFTick(gainValue, gainEnvValue_, attackCoef);
148  }
149  else {
150  onePoleLPFTick(gainValue, gainEnvValue_, releaseCoef);
151  }
152 
153  // apply gain
154  for (unsigned int i=0; i<nChannels; i++){
155  *outptr++ = lookaheadDelayLine_.tickOut(lookaheadTime,i) * gainEnvValue_;
156  }
157 
158  lookaheadDelayLine_.advance();
159  }
160 
161  TonicFloat makeupGain = max(0.f, makeupGainGen_.tick(context).value);
162  outptr = &outputFrames_[0];
163 
164  #ifdef USE_APPLE_ACCELERATE
165  vDSP_vsmul(outptr, 1, &makeupGain, outptr, 1, outputFrames_.size());
166  #else
167  for (unsigned int i=0; i<outputFrames_.size(); i++){
168  *outptr++ *= makeupGain;
169  }
170  #endif
171 
172  if (isLimiter_){
173 
174  // clip to threshold in worst case (minor distortion introduced but much preferable to wrapping distortion)
175  #ifdef USE_APPLE_ACCELERATE
176  float negThresh = -threshold;
177  vDSP_vclip(&outputFrames_[0], 1, &negThresh, &threshold, &outputFrames_[0], 1, outputFrames_.size());
178  #else
179  outptr = &outputFrames_[0];
180  for (unsigned int i=0; i<outputFrames_.size(); i++, outptr++){
181  *outptr = clamp(*outptr, -threshold, threshold);
182  }
183  #endif
184 
185  }
186 
187  }
