53  inline void AngularWave_::computeSynthesisBlock(const SynthesisContext_ &context){
54 
55  // tick freq and slope inputs
56  freqGen_.tick(freqFrames_, context);
57  slopeGen_.tick(slopeFrames_, context);
58 
59  // calculate the output wave
60  TonicFloat const rateConstant = TONIC_SAW_RES/Tonic::sampleRate();
61 
62  TonicFloat slope, frac, phase;
63  TonicFloat *outptr = &outputFrames_[0];
64  TonicFloat *freqptr = &freqFrames_[0];
65  TonicFloat *slopeptr = &slopeFrames_[0];
66 
67  ShiftedDouble sd;
68 
69  // pre-multiply rate constant for speed
70 #ifdef USE_APPLE_ACCELERATE
71  vDSP_vsmul(freqptr, 1, &rateConstant, freqptr, 1, kSynthesisBlockSize);
72 #else
73  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
74  *freqptr++ *= rateConstant;
75  }
76  freqptr = &freqFrames_[0];
77 #endif
78 
79  sd.d = BIT32DECPT;
80  TonicInt32 offs, msbi = sd.i[1];
81  double ps = phaseAccum_ + BIT32DECPT;
82  for ( unsigned int i=0; i<outputFrames_.frames(); i++ ) {
83 
84  // update the slope
85  slope = clamp(*slopeptr++, 0.0f, 1.0f) * TONIC_SAW_RES;
86 
87  sd.d = ps;
88  ps += *freqptr++;
89  offs = sd.i[1] & (TONIC_SAW_RES-1);
90  sd.i[1] = msbi;
91  frac = sd.d - BIT32DECPT;
92 
93  phase = offs + frac;
94 
95  // TODO: this is still a bit slow. Maybe find a way to do it with a real table lookup
96 
97  // on the rising edge
98  if (phase < slope){
99  *outptr++ = (phase/slope)*2.0f - 1.0f;
100  }
101  // on the falling edge
102  else{
103  *outptr++ = (1.0f - ((phase - slope)/(TONIC_SAW_RES - slope))) * 2.0f - 1.0f;
104  }
105 
106  }
107 
108  sd.d = BIT32DECPT * TONIC_SAW_RES;
109  msbi = sd.i[1];
110  sd.d = ps + (BIT32DECPT * TONIC_SAW_RES - BIT32DECPT);
111  sd.i[1] = msbi;
112  phaseAccum_ = sd.d - BIT32DECPT * TONIC_SAW_RES;
113 
114  }
