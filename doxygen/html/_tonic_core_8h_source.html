184  inline bool isPowerOf2(unsigned int input, unsigned int * prevPo2){
185 
186  if (input == 0) return true;
187 
188  unsigned int po2 = 2;
189  while (po2 < input){
190  po2 *= 2;
191  }
192 
193  if (prevPo2){
194 
195  unsigned int nextPo2 = po2 * 2;
196  *prevPo2 = abs((int)input - (int)po2) < abs((int)input - (int)nextPo2) ? po2 : nextPo2;
197  }
198 
199  return input == po2;
200  }
