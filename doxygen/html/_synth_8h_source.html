152  struct SynthFactory {

154  typedef std::map<std::string, Synth(*)()> map_type;
155 
156  static Synth createInstance(std::string const& s) {
157  map_type::iterator it = getMap()->find(s);
158  if(it == getMap()->end()){
159  string synthsList = "";
160 
161  for(it = getMap()->begin(); it != getMap()->end(); it++){
162  synthsList = synthsList + it->first + "\n";
163  }
164 
165  error("Error creating synth. Synth named \"" + s + "\" not found. Existing registered synths are: \n" + synthsList);
166 
167  // return empty synth
168  return Synth();
169  }
170  return it->second();
171  }
172 
173  protected:
174  static map_type * getMap() {
175  // never delete'ed. (exist until program termination)
176  // because we can't guarantee correct destruction order
177  if(!map) { map = new map_type; }
