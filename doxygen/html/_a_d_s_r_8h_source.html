88  inline void ADSR_::computeSynthesisBlock(const SynthesisContext_ &context){
89 
90  ControlGeneratorOutput triggerOutput = mTrigger.tick(context);
91 
92  // frames to go in this block
93 
94  // Tick ALL inputs every time to keep everything in sync
95  attackTime = attack.tick(context).value;
96  decayTime = decay.tick(context).value;
97  sustainLevelVal = sustain.tick(context).value;
98  releaseTime = release.tick(context).value;
99  bIsExponential = (bool)isExponential.tick(context).value;
100  bDoesSustain = (bool)doesSustain.tick(context).value;
101  bIsLegato = (bool)isLegato.tick(context).value;
102 
103  TonicFloat * fdata = &outputFrames_[0];
104 
105  if(triggerOutput.status == ControlGeneratorStatusHasChanged){
106 
107  if(triggerOutput.value != 0){
108  switchState(ATTACK);
109  }else if(bDoesSustain){
110  switchState(RELEASE);
111  }
112 
113  }
114 
115  int samplesRemaining = kSynthesisBlockSize;
116 
117  while (samplesRemaining > 0)
118  {
119  switch (state) {
120 
121  // Both of these cases just fill the synthesis block the rest of the way up
122  case NEUTRAL:
123  case SUSTAIN:
124  {
125  #ifdef USE_APPLE_ACCELERATE
126  vDSP_vfill(&lastValue, fdata, 1, samplesRemaining);
127  #else
128  std::fill(fdata, fdata + samplesRemaining, lastValue);
129  #endif
130 
131  samplesRemaining = 0;
132  }
133  break;
134 
135  case ATTACK:
136  case DECAY:
137  case RELEASE:
138  {
139 
140  // how many samples remain in current segment
141  unsigned long remainder = (segCounter > segLength) ? 0 : segLength - segCounter;
142  if (remainder < samplesRemaining){
143 
144  // fill up part of the ramp then switch segment
145 
146  if (bIsExponential){
147 
148  // one pole filter
149  for (unsigned long i=0; i<remainder; i++){
150  onePoleLPFTick(targetValue, lastValue, pole);
151  *fdata++ = lastValue;
152  }
153 
154  }
155  else{
156 
157  #ifdef USE_APPLE_ACCELERATE
158  // starting point
159  lastValue += increment;
160 
161  // vector calculation
162  vDSP_vramp(&lastValue, &increment, fdata, 1, remainder);
163 
164  // end point
165  lastValue += increment*(remainder-1);
166  fdata += remainder;
167 
168  #else
169  for (unsigned long i=0; i<remainder; i++){
170  lastValue += increment;
171  *fdata++ = lastValue;
172  }
173  #endif
174  }
175 
176  segCounter += remainder;
177  samplesRemaining -= remainder;
178 
179  // switch segment
180  if (state == ATTACK){
181  switchState(DECAY);
182  }
183  else if (state == DECAY){
184  switchState(bDoesSustain ? SUSTAIN : RELEASE);
185  }
186  else{
187  switchState(NEUTRAL);
188  }
189 
190  }
191  else{
192 
193  if (bIsExponential){
194 
195  // one pole filter
196  for (int i=0; i<samplesRemaining; i++){
197  onePoleLPFTick(targetValue, lastValue, pole);
198  *fdata++ = lastValue;
199  }
200 
201  }
202  else{
203  // fill the rest of the ramp up
204  #ifdef USE_APPLE_ACCELERATE
205  // starting point
206  lastValue += increment;
207 
208  // vector calculation
209  vDSP_vramp(&lastValue, &increment, fdata, 1, samplesRemaining);
210 
211  // end point
212  lastValue += increment*(samplesRemaining-1);
213 
214  #else
215  for (int i=0; i<samplesRemaining; i++){
216  lastValue += increment;
217  *fdata++ = lastValue;
218  }
219  #endif
220 
221  }
222 
223  segCounter += samplesRemaining;
224  samplesRemaining = 0;
225  }
226  }
227  break;
