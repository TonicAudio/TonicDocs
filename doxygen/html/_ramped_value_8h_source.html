54  inline void RampedValue_::computeSynthesisBlock( const SynthesisContext_ & context ){
55 
56  // First set the value, if necessary (abort ramp, go immediately to value)
57  ControlGeneratorOutput valueOutput = valueGen_.tick(context);
58  if(valueOutput.status == ControlGeneratorStatusHasChanged){
59  updateValue(valueOutput.value);
60  }
61 
62  // Then update the target or ramp length (start a new ramp)
63  ControlGeneratorOutput lengthOutput = lengthGen_.tick(context);
64  ControlGeneratorOutput targetOutput = targetGen_.tick(context);
65  if (lengthOutput.status == ControlGeneratorStatusHasChanged || targetOutput.status == ControlGeneratorStatusHasChanged){
66  unsigned long lSamp = lengthOutput.value*Tonic::sampleRate();
67  updateTarget(targetOutput.value, lSamp);
68  }
69 
70  TonicFloat *fdata = &outputFrames_[0];
71  unsigned int nFrames = kSynthesisBlockSize;
72  unsigned int stride = outputFrames_.channels();
73 
74  // edge case
75  if (count_ == len_){
76  finished_ = true;
77  }
78 
79  if (finished_){
80  #ifdef USE_APPLE_ACCELERATE
81  vDSP_vfill(&last_, fdata, stride, nFrames);
82  #else
83  for (unsigned int i=0; i<nFrames; i++){
84  *fdata = last_;
85  fdata += stride;
86  }
87  #endif
88  }
89  else{
90 
91  // figure out if we will finish the ramp in this tick
92  unsigned long remainder = count_ > len_ ? 0 : len_ - count_;
93 
94  if (remainder < nFrames){
95 
96  // fill part of the ramp - no need to check if remainder is 1 when using vramp
97  // uses algorithm out[n] = start + n * increment;
98  #ifdef USE_APPLE_ACCELERATE
99  // starting point
100  last_ += inc_;
101  vDSP_vramp(&last_, &inc_, fdata, stride, remainder);
102 
103  #ifdef TONIC_DEBUG
104  if(*fdata != *fdata){
105  Tonic::error("RampedValue_::computeSynthesisBlock NaN detected.\n");
106  }
107  #endif
108 
109  #else
110  for (unsigned int i=0; i<remainder; i++){
111  last_ += inc_;
112  *fdata = last_;
113  fdata += stride;
114  }
115  #endif
116 
117  #ifdef USE_APPLE_ACCELERATE
118  vDSP_vfill(&target_, fdata + remainder, stride, nFrames - remainder);
119  #else
120  for (unsigned int i=remainder; i<nFrames; i++){
121  *fdata = target_;
122  fdata += stride;
123  }
124  #endif
125 
126  count_ = len_;
127  last_ = target_;
128  finished_ = true;
129  }
130  else{
131 
132  // fill the whole ramp
133  #ifdef USE_APPLE_ACCELERATE
134  last_ += inc_;
135  vDSP_vramp(&last_, &inc_, fdata, stride, nFrames);
136 
137  #ifdef TONIC_DEBUG
138  if(*fdata != *fdata){
139  Tonic::error("RampedValue_::computeSynthesisBlock NaN detected.\n");
140  }
141  #endif
142 
143  #else
144  for (unsigned int i=0; i<nFrames; i++){
145  last_ += inc_;
146  *fdata = last_;
147  fdata += stride;
148  }
149  #endif
150

153  }
154  }
