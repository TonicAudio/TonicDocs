58  inline void ImpulseDiffuserAllpass::tickThrough(Tonic::TonicFrames &frames)
59  {
60  TonicFloat *dptr = &frames[0];
61  TonicFloat y;
62  for (int i=0; i<kSynthesisBlockSize; i++){
63 
64  // feedback stage
65  y = *dptr + delayBack_.tickOut(delay_) * coef_;
66  delayBack_.tickIn(y);
67  delayBack_.advance();
68 
69  // feed forward stage
70  *dptr++ = (1.f+coef_)*delayForward_.tickOut(delay_) - y;
71  delayForward_.tickIn(y);
72  delayForward_.advance();
73  }
74  }

151  inline void Reverb_::computeSynthesisBlock(const SynthesisContext_ &context){
152 
153  updateDelayTimes(context);
154 
155  // pass thru input filters
156  if (inputFiltBypasCtrlGen_.tick(context).value == 0.f){
157 
158  inputLPF_.tickThrough(dryFrames_, workspaceFrames_[0], context);
159  inputHPF_.tickThrough(workspaceFrames_[0], workspaceFrames_[0], context);
160 
161  }
162  else{
163  workspaceFrames_[0].copy(dryFrames_);
164  }
165 
166  TonicFloat *wkptr0 = &(workspaceFrames_[0])[0];
167  TonicFloat *wkptr1 = &(workspaceFrames_[1])[0];
168 
169  // pass thru pre-delay, input filters, and sum the early reflections
170 
171  TonicFloat preDelayTime = preDelayTimeCtrlGen_.tick(context).value;
172 
173  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
174 
175  // filtered input is in w0
176  // predelay output is in w1
177 
178  // pre-delay
179  preDelayLine_.tickIn(*wkptr0);
180  *wkptr1 = preDelayLine_.tickOut(preDelayTime);
181  preDelayLine_.advance();
182 
183  // taps - write back to w0
184  reflectDelayLine_.tickIn(*wkptr1++);
185  *wkptr0 = 0;
186  for (unsigned int t=0; t<reflectTapTimes_.size(); t++){
187  *wkptr0 += reflectDelayLine_.tickOut(reflectTapTimes_[t]) * reflectTapScale_[t];
188  }
189 
190  reflectDelayLine_.advance();
191  wkptr0++;
192  }
193 
194  // Comb filers
195  preOutputFrames_[TONIC_LEFT].clear();
196  preOutputFrames_[TONIC_RIGHT].clear();
197  for (unsigned int i=0; i<combFilters_[TONIC_LEFT].size(); i++){
198  combFilters_[TONIC_LEFT][i].tickThrough(workspaceFrames_[0], workspaceFrames_[1], context);
199  preOutputFrames_[TONIC_LEFT] += workspaceFrames_[1];
200  combFilters_[TONIC_RIGHT][i].tickThrough(workspaceFrames_[0], workspaceFrames_[1], context);
201  preOutputFrames_[TONIC_RIGHT] += workspaceFrames_[1];
202  }
203 
204  // Allpass filters
205  for (unsigned int i=0; i<allpassFilters_[TONIC_LEFT].size(); i++){
206  allpassFilters_[TONIC_LEFT][i].tickThrough(preOutputFrames_[TONIC_LEFT]);
207  allpassFilters_[TONIC_RIGHT][i].tickThrough(preOutputFrames_[TONIC_RIGHT]);
208  }
209 
210  // interleave pre-output frames into output frames
211  TonicFloat *outptr = &outputFrames_[0];
212  TonicFloat *preoutptrL = &preOutputFrames_[TONIC_LEFT][0];
213  TonicFloat *preoutptrR = &preOutputFrames_[TONIC_RIGHT][0];
214 
215  TonicFloat spreadValue = clamp(1.0f - stereoWidthCtrlGen_.tick(context).value, 0.f, 1.f);
216  TonicFloat normValue = (1.0f/(1.0f+spreadValue))*0.04f; // scale back levels quite a bit
217 
218  for (unsigned int i=0; i<kSynthesisBlockSize; i++){
219  *outptr++ = (*preoutptrL + (spreadValue * (*preoutptrR)))*normValue;
220  *outptr++ = (*preoutptrR++ + (spreadValue * (*preoutptrL++)))*normValue;
221  }
222 
223 
224  }
