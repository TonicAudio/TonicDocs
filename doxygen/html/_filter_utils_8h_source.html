52  inline static void bltCoef( TonicFloat b2, TonicFloat b1, TonicFloat b0, TonicFloat a1, TonicFloat a0, TonicFloat fc, TonicFloat *coef_out)
53  {
54  TonicFloat sf = 1.0f/tanf(PI*fc/Tonic::sampleRate());
55  TonicFloat sfsq = sf*sf;
56  TonicFloat norm = a0 + a1*sf + sfsq;
57  coef_out[0] = (b0 + b1*sf + b2*sfsq)/norm;
58  coef_out[1] = 2.0f * (b0 - b2*sfsq)/norm;
