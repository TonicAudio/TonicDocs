87 
88  // check context to see if we need new frames
89  if (context.elapsedFrames == 0 || lastFrameIndex_ != context.elapsedFrames){
90 
91  // get dry input frames
92  input_.tick(dryFrames_, context);
93 
94  computeSynthesisBlock(context);
95

97  bool bypass = bypassGen_.tick(context).value != 0.f;
98  if (bypass){

100  }
101  else if (!isAlwaysWet_){
102  // do not apply dry/wet levels if always wet flag is set
103  // offers minor CPU usage optimization
104  wetLevelGen_.tick(mixWorkspace_, context);
105  outputFrames_ *= mixWorkspace_;
106  dryLevelGen_.tick(mixWorkspace_, context);
107  dryFrames_ *= mixWorkspace_;
108  outputFrames_ += dryFrames_;
109  }

112  }
